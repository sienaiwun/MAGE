522  bool Collides(const Sphere &sphere, const XMFLOAT3 velocity_sum, float *collision_distance) {
523  const XMVECTOR p1_v = XMLoadFloat3(&p);
524  const XMVECTOR p2_v = XMLoadFloat3(&sphere.p);
525 
526  // Calculate the direction vector from the second sphere to the first sphere.
527  const XMVECTOR direction_v = p1_v - p2_v;
528 
529  // Calculate the distance between the two spheres.
530  const XMVECTOR dist_v = XMVector3Length(direction_v);
531  float dist;
532  XMStoreFloat(&dist, dist_v);
533  const float radii_sum = r + sphere.r;
534  const float dist_between = dist - radii_sum;
535 
536  // Calculate the length of the sum of the velocity vectors of the two spheres.
537  const XMVECTOR velocity_sum_v = XMLoadFloat3(&velocity_sum);
538  const XMVECTOR velocity_sum_length_v = XMVector3Length(velocity_sum_length_v);
539  float velocity_sum_length;
540  XMStoreFloat(&velocity_sum_length, velocity_sum_length_v);
541 
542  // If the spheres are not touching each other and the velocity sum length is
543  // less than the distance between them, then they cannot collide.
544  if (0.0f < dist_between && velocity_sum_length < dist_between) {
545  return false;
546  }
547 
548  // Calculate the normalized sum of the velocity vectors of the two spheres.
549  const XMVECTOR velocity_sum_normalized_v = XMVector3Normalize(velocity_sum_v);
550 
551  // Calculate the angle between the normalized sum of the velocity vectors and direction vectors.
552  const XMVECTOR angle_between_v = XMVector3Dot(velocity_sum_normalized_v, direction_v);
553  float angle_between;
554  XMStoreFloat(&angle_between, angle_between_v);
555 
556  // Check whether the spheres are moving away from one another.
557  if (angle_between <= 0.0f) {
558  // Check whether the spheres are touching (or inside) each other.
559  // If not then they cannot collide since they are moving away from one another.
560  if (dist_between < 0.0f) {
561  // If the velocity sum length is greater than the distance between the
562  // spheres then they are moving away from each other fast enough that
563  // they will not be touching when they complete their move.
564  if (velocity_sum_length > -dist_between) {
565  return false;
566  }
567  }
568  else {
569  return false;
570  }
571  }
572 
573  // The vector between the two spheres and the velocity sum vector produce two sides of a triangle.
574  // Now use Pythagorean Theorem to find the length of the third side of the triangle (i.e. the hypotenuse).
575  const float hypotenuse = (dist * dist) - (angle_between * angle_between);
576 
577  // Ensure that the spheres come closer than the sum of their radii.
578  const float squared_radii_sum = radii_sum * radii_sum;
579  if (hypotenuse >= squared_radii_sum) {
580  return false;
581  }
582 
583  // Calculate the distance along the velocity vector that the spheres collide.
584  // Then use this distance to calculate the distance to the collision.
585  const float d = squared_radii_sum - hypotenuse;
586  *collision_distance = angle_between - (float)sqrt(d);
587 
588  // Ensure that the sphere will not travel more than the velocity allows.
589  return (velocity_sum_length >= *collision_distance);
590  }
