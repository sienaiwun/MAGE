561  bool Collides(const Sphere &sphere, const XMFLOAT3 velocity_sum, float *collision_distance) {
562  const XMVECTOR p1_v = XMLoadFloat3(&p);
563  const XMVECTOR p2_v = XMLoadFloat3(&sphere.p);
564 
565  // Calculate the direction vector from the second sphere to the first sphere.
566  const XMVECTOR direction_v = p1_v - p2_v;
567 
568  // Calculate the distance between the two spheres.
569  const XMVECTOR dist_v = XMVector3Length(direction_v);
570  float dist;
571  XMStoreFloat(&dist, dist_v);
572  const float radii_sum = r + sphere.r;
573  const float dist_between = dist - radii_sum;
574 
575  // Calculate the length of the sum of the velocity vectors of the two spheres.
576  const XMVECTOR velocity_sum_v = XMLoadFloat3(&velocity_sum);
577  const XMVECTOR velocity_sum_length_v = XMVector3Length(velocity_sum_length_v);
578  float velocity_sum_length;
579  XMStoreFloat(&velocity_sum_length, velocity_sum_length_v);
580 
581  // If the spheres are not touching each other and the velocity sum length is
582  // less than the distance between them, then they cannot collide.
583  if (0.0f < dist_between && velocity_sum_length < dist_between) {
584  return false;
585  }
586 
587  // Calculate the normalized sum of the velocity vectors of the two spheres.
588  const XMVECTOR velocity_sum_normalized_v = XMVector3Normalize(velocity_sum_v);
589 
590  // Calculate the angle between the normalized sum of the velocity vectors and direction vectors.
591  const XMVECTOR angle_between_v = XMVector3Dot(velocity_sum_normalized_v, direction_v);
592  float angle_between;
593  XMStoreFloat(&angle_between, angle_between_v);
594 
595  // Check whether the spheres are moving away from one another.
596  if (angle_between <= 0.0f) {
597  // Check whether the spheres are touching (or inside) each other.
598  // If not then they cannot collide since they are moving away from one another.
599  if (dist_between < 0.0f) {
600  // If the velocity sum length is greater than the distance between the
601  // spheres then they are moving away from each other fast enough that
602  // they will not be touching when they complete their move.
603  if (velocity_sum_length > -dist_between) {
604  return false;
605  }
606  }
607  else {
608  return false;
609  }
610  }
611 
612  // The vector between the two spheres and the velocity sum vector produce two sides of a triangle.
613  // Now use Pythagorean Theorem to find the length of the third side of the triangle (i.e. the hypotenuse).
614  const float hypotenuse = (dist * dist) - (angle_between * angle_between);
615 
616  // Ensure that the spheres come closer than the sum of their radii.
617  const float squared_radii_sum = radii_sum * radii_sum;
618  if (hypotenuse >= squared_radii_sum) {
619  return false;
620  }
621 
622  // Calculate the distance along the velocity vector that the spheres collide.
623  // Then use this distance to calculate the distance to the collision.
624  const float d = squared_radii_sum - hypotenuse;
625  *collision_distance = angle_between - (float)sqrt(d);
626 
627  // Ensure that the sphere will not travel more than the velocity allows.
628  return (velocity_sum_length >= *collision_distance);
629  }
