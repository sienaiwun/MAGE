331  Transform &SetRotationAroundDirection(const XMVECTOR &normal, float angle) {
332  const XMMATRIX rotation_m = XMMatrixRotationNormal(normal, angle);
333  XMFLOAT4X4 rotation;
334  XMStoreFloat4x4(&rotation, rotation_m);
335 
336  // cosf function instead of sinf in case the angles are greater than [-1,1]
337  m_rotation.y = -asinf(rotation._32);
338  const float cp = cosf(m_rotation.y);
339  const float cr = rotation._22 / cp;
340  m_rotation.z = acosf(cr);
341  const float cy = rotation._33 / cp;
342  m_rotation.x = acosf(cy);
343 
344  return (*this);
345  }
