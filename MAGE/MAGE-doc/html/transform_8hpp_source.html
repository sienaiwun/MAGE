341  Transform &SetRotationAroundDirection(const XMVECTOR &normal, float angle) {
342  const XMMATRIX rotation_m = XMMatrixRotationNormal(normal, angle);
343  XMFLOAT4X4 rotation;
344  XMStoreFloat4x4(&rotation, rotation_m);
345 
346  // cosf function instead of sinf in case the angles are greater than [-1,1]
347  m_rotation.y = -asinf(rotation._32);
348  const float cp = cosf(m_rotation.y);
349  const float cr = rotation._22 / cp;
350  m_rotation.z = acosf(cr);
351  const float cy = rotation._33 / cp;
352  m_rotation.x = acosf(cy);
353 
354  return (*this);
355  }
