682  bool Collides(const BS &sphere, const XMFLOAT3 velocity_sum, float *collision_distance) {
683  const XMVECTOR p1_v = XMLoadFloat3(&p);
684  const XMVECTOR p2_v = XMLoadFloat3(&sphere.p);
685 
686  // Calculate the direction vector from the second sphere to the first sphere.
687  const XMVECTOR direction_v = p1_v - p2_v;
688 
689  // Calculate the distance between the two spheres.
690  const XMVECTOR dist_v = XMVector3Length(direction_v);
691  float dist;
692  XMStoreFloat(&dist, dist_v);
693  const float radii_sum = r + sphere.r;
694  const float dist_between = dist - radii_sum;
695 
696  // Calculate the length of the sum of the velocity vectors of the two spheres.
697  const XMVECTOR velocity_sum_v = XMLoadFloat3(&velocity_sum);
698  const XMVECTOR velocity_sum_length_v = XMVector3Length(velocity_sum_length_v);
699  float velocity_sum_length;
700  XMStoreFloat(&velocity_sum_length, velocity_sum_length_v);
701 
702  // If the spheres are not touching each other and the velocity sum length is
703  // less than the distance between them, then they cannot collide.
704  if (0.0f < dist_between && velocity_sum_length < dist_between) {
705  return false;
706  }
707 
708  // Calculate the normalized sum of the velocity vectors of the two spheres.
709  const XMVECTOR velocity_sum_normalized_v = XMVector3Normalize(velocity_sum_v);
710 
711  // Calculate the angle between the normalized sum of the velocity vectors and direction vectors.
712  const XMVECTOR angle_between_v = XMVector3Dot(velocity_sum_normalized_v, direction_v);
713  float angle_between;
714  XMStoreFloat(&angle_between, angle_between_v);
715 
716  // Check whether the spheres are moving away from one another.
717  if (angle_between <= 0.0f) {
718  // Check whether the spheres are touching (or inside) each other.
719  // If not then they cannot collide since they are moving away from one another.
720  if (dist_between < 0.0f) {
721  // If the velocity sum length is greater than the distance between the
722  // spheres then they are moving away from each other fast enough that
723  // they will not be touching when they complete their move.
724  if (velocity_sum_length > -dist_between) {
725  return false;
726  }
727  }
728  else {
729  return false;
730  }
731  }
732 
733  // The vector between the two spheres and the velocity sum vector produce two sides of a triangle.
734  // Now use Pythagorean Theorem to find the length of the third side of the triangle (i.e. the hypotenuse).
735  const float hypotenuse = (dist * dist) - (angle_between * angle_between);
736 
737  // Ensure that the spheres come closer than the sum of their radii.
738  const float squared_radii_sum = radii_sum * radii_sum;
739  if (hypotenuse >= squared_radii_sum) {
740  return false;
741  }
742 
743  // Calculate the distance along the velocity vector that the spheres collide.
744  // Then use this distance to calculate the distance to the collision.
745  const float d = squared_radii_sum - hypotenuse;
