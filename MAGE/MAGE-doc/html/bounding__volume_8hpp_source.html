672  bool Collides(const BS &sphere, const XMFLOAT3 velocity_sum, float *collision_distance) {
673  const XMVECTOR p1_v = XMLoadFloat3(&p);
674  const XMVECTOR p2_v = XMLoadFloat3(&sphere.p);
675 
676  // Calculate the direction vector from the second sphere to the first sphere.
677  const XMVECTOR direction_v = p1_v - p2_v;
678 
679  // Calculate the distance between the two spheres.
680  const XMVECTOR dist_v = XMVector3Length(direction_v);
681  float dist;
682  XMStoreFloat(&dist, dist_v);
683  const float radii_sum = r + sphere.r;
684  const float dist_between = dist - radii_sum;
685 
686  // Calculate the length of the sum of the velocity vectors of the two spheres.
687  const XMVECTOR velocity_sum_v = XMLoadFloat3(&velocity_sum);
688  const XMVECTOR velocity_sum_length_v = XMVector3Length(velocity_sum_length_v);
689  float velocity_sum_length;
690  XMStoreFloat(&velocity_sum_length, velocity_sum_length_v);
691 
692  // If the spheres are not touching each other and the velocity sum length is
693  // less than the distance between them, then they cannot collide.
694  if (0.0f < dist_between && velocity_sum_length < dist_between) {
695  return false;
696  }
697 
698  // Calculate the normalized sum of the velocity vectors of the two spheres.
699  const XMVECTOR velocity_sum_normalized_v = XMVector3Normalize(velocity_sum_v);
700 
701  // Calculate the angle between the normalized sum of the velocity vectors and direction vectors.
702  const XMVECTOR angle_between_v = XMVector3Dot(velocity_sum_normalized_v, direction_v);
703  float angle_between;
704  XMStoreFloat(&angle_between, angle_between_v);
705 
706  // Check whether the spheres are moving away from one another.
707  if (angle_between <= 0.0f) {
708  // Check whether the spheres are touching (or inside) each other.
709  // If not then they cannot collide since they are moving away from one another.
710  if (dist_between < 0.0f) {
711  // If the velocity sum length is greater than the distance between the
712  // spheres then they are moving away from each other fast enough that
713  // they will not be touching when they complete their move.
714  if (velocity_sum_length > -dist_between) {
715  return false;
716  }
717  }
718  else {
719  return false;
720  }
721  }
722 
723  // The vector between the two spheres and the velocity sum vector produce two sides of a triangle.
724  // Now use Pythagorean Theorem to find the length of the third side of the triangle (i.e. the hypotenuse).
725  const float hypotenuse = (dist * dist) - (angle_between * angle_between);
726 
727  // Ensure that the spheres come closer than the sum of their radii.
728  const float squared_radii_sum = radii_sum * radii_sum;
729  if (hypotenuse >= squared_radii_sum) {
730  return false;
731  }
732 
733  // Calculate the distance along the velocity vector that the spheres collide.
734  // Then use this distance to calculate the distance to the collision.
735  const float d = squared_radii_sum - hypotenuse;
736  *collision_distance = angle_between - (float)sqrt(d);
737 
738  // Ensure that the sphere will not travel more than the velocity allows.
739  return (velocity_sum_length >= *collision_distance);
740  }
